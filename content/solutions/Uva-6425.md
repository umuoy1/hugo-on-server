知识点：BFS，链表思想

### 题目大意

给出一张完全图，每条边边权为b，现在把其中m条边的边权替换为a，问从1到n的最短路径是多少。

### 分析

没啥好说的，分两种情况bfs，第一种是只走新的边，第二种是只走旧的边。不能混走，可以证明在走单种边的情况下才会有最短路，给出简易证明：假设最短路a--b--c的边a--b为第一种边，b--c为第二种边，但由于图是完全图，有边a--c，无论a--c为哪种边，`dis(a,b)+dis(b,c)>len(a,c)`必定成立，与假设矛盾，得证。

在第二种情况bfs时，可以设一个变量，表示当前未走过的点和已经走过的点（这里用的是链表，走过的点直接删除），在此基础上bfs。

### 代码

```c++
#include<iostream>
#include<cstdio>
#include<queue>
#include<cstring>
#include<set>
#define INF 1000000000
#define N 500005
using namespace std;
int n,k;
long long a,b;
int dis[N],head[N],num;
set<int>s[N];
struct edge{
    int to,nex;
}e[N*2];
void ins(int x,int y){
    e[++num].nex=head[x];
    e[num].to=y;
    head[x]=num;
}
long long bfs1(){
    queue<int>q;
    q.push(1);
    memset(dis,-1,sizeof(dis));
    dis[1]=0;
    while(!q.empty()){
        int x = q.front();
        q.pop();
        for(int i=head[x];i;i=e[i].nex){
            int v = e[i].to;
            if(dis[v]!=-1) continue;
            dis[v]=dis[x]+1;
            q.push(v);
            if(v==n)
                return dis[v];
        }
    }
    return INF;
}
int L[N],R[N];
long long bfs2(){
    for(int i=1;i<=n;i++)
        L[i]=i-1,R[i]=i+1;
    queue<int>q;
    q.push(1);
    if(s[1].size()==n-1)
        return INF;
    memset(dis,-1,sizeof(dis));
    dis[1]=0;
    while(!q.empty()){
        int x=q.front();
        q.pop();
        for(int v=R[1];v!=n+1;v=R[v]){
            if(s[x].count(v)) continue;
            dis[v]=dis[x]+1;q.push(v);
            if(v==n)
                return dis[v];
            L[R[v]]=L[v];R[L[v]]=R[v];
        }
    }
    return INF;
}
int main(){
    cin>>n>>k>>a>>b;
    for(int i=1;i<=k;i++){
        int x,y;
        scanf("%d %d",&x,&y);
        ins(x,y);
        ins(y,x);
        s[x].insert(y);
        s[y].insert(x);
    }
    if(a==b){
        cout<<a;
    }
    else{
        long long ans1=bfs1();
        long long ans2=bfs2();
        cout<<min(ans1*a,ans2*b);
    }
    
}
```
